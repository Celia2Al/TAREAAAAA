package Ejercicio1;

public class Factura {
        // Variables de instancia
        private String numeroPieza;
        private String descripcionPieza;
        private int cantidad;
        private double precioPorArticulo;

        // Constructor que inicializa las variables
        public Factura(String numeroPieza, String descripcionPieza, int cantidad, double precioPorArticulo) {
            this.numeroPieza = numeroPieza;
            this.descripcionPieza = descripcionPieza;
            setCantidad(cantidad); // usamos el setter para aplicar validación
            setPrecioPorArticulo(precioPorArticulo); // usamos el setter para aplicar validación
        }

        // Métodos set (establecer)
        public void setNumeroPieza(String numeroPieza) {
            this.numeroPieza = numeroPieza;
        }

        public void setDescripcionPieza(String descripcionPieza) {
            this.descripcionPieza = descripcionPieza;
        }

        public void setCantidad(int cantidad) {
            // Validación: si es negativa, se pone en 0
            if (cantidad > 0) {
                this.cantidad = cantidad;
            } else {
                this.cantidad = 0;
            }
        }

        public void setPrecioPorArticulo(double precioPorArticulo) {
            // Validación: si es negativa, se pone en 0.0
            if (precioPorArticulo > 0.0) {
                this.precioPorArticulo = precioPorArticulo;
            } else {
                this.precioPorArticulo = 0.0;
            }
        }

        // Métodos get (obtener)
        public String getNumeroPieza() {
            return numeroPieza;
        }

        public String getDescripcionPieza() {
            return descripcionPieza;
        }

        public int getCantidad() {
            return cantidad;
        }

        public double getPrecioPorArticulo() {
            return precioPorArticulo;
        }

        // Método que calcula el monto total de la factura
        public double obtenerMontoFactura() {
            return cantidad * precioPorArticulo;
        }
    }
